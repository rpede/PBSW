\section{Q 1.4: Conducting a usability test}

\begin{frame}
    \frametitle{5 E's}
    \begin{itemize}
        \item Effective
            % How completely and accurately the work or experience is completed
            % or goals reached.
        \item Efficient
            % How quickly this work can be done.
        \item Engaging
            % How well the interface draws the user into the interaction and
            % how pleasant and satisfying it is to use.
        \item Error Tolerant
            % How well the product prevents errors and can help the user
            % recover from mistakes that do occur.
        \item Easy to learn
            % How well the product supports both the initial orientation and
            % continued learning throughout the complete lifetime of use.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Think Aloud Protocol}
    Have the user tell about their thoughts, impressions, considerations and
    actions.
    % Use verbal communication as a windows to the users brain
\end{frame}

\begin{frame}
    \frametitle{Participants}
    \begin{itemize}
        \item Moderator
            % Conduct/lead the test. Briefing, debriefing etc.
        \item Scriber
            % Record observations
        \item Observer
            % Take notes
        \item User
            % Represents the end user
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Conducing a Usability Test}
    \begin{enumerate}
        \item Planning
            \begin{itemize}
                \item Place, Time
                \item Test cases
            \end{itemize}
        \item Executing
            \begin{itemize}
                \item Briefing
                \item Execution
                \item Debriefing
            \end{itemize}
        \item Reporting
            \begin{itemize}
                \item Collecting/Organize
                \item Aggregate/Summarize
                \item Analyze
            \end{itemize}
    \end{enumerate}
\end{frame}
